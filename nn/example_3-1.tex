\def\pointTop(#1)#2#3{%
	\pscircle[
		linewidth = 1.7\pslinewidth
	](#1){0.35}
	\rput(#1){\textcolor{red}{$c_{#2}$}}
	\uput[90](#1){$h_{#3}$}
}

\def\pointBottom(#1)#2#3{%
	\pscircle[
		linewidth = 1.7\pslinewidth,
		fillstyle = solid,
		fillcolor = gray!50
	](#1){0.35}
	\rput(#1){\textcolor{red}{$b_{#2}$}}
	\uput[270](#1){$v_{#3}$}
}
